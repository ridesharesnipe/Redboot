```jsx
// WeeklyActivityChart.jsx

import React from 'react';

const WeeklyActivityChart = ({ sessions }) => {
  // Get last 7 days
  const getLast7Days = () => {
    const days = [];
    const dayNames = ['SUN', 'MON', 'TUE', 'WED', 'THU', 'FRI', 'SAT'];
    
    for (let i = 6; i >= 0; i--) {
      const date = new Date();
      date.setDate(date.getDate() - i);
      days.push({
        date: date.toISOString().split('T')[0],
        dayName: dayNames[date.getDay()],
        isToday: i === 0,
      });
    }
    return days;
  };

  // Calculate minutes practiced per day from REAL session data
  const getMinutesPerDay = (targetDate) => {
    const daySessions = sessions.filter(session => {
      const sessionDate = new Date(session.timestamp).toISOString().split('T')[0];
      return sessionDate === targetDate;
    });
    
    const totalSeconds = daySessions.reduce((sum, s) => sum + (s.durationSeconds || 0), 0);
    return Math.round(totalSeconds / 60);
  };

  const days = getLast7Days();
  const maxMinutes = Math.max(...days.map(d => getMinutesPerDay(d.date)), 1);

  return (
    <div className="weekly-activity-container">
      <div className="weekly-activity-header">
        <div>
          <h3>Practice This Week</h3>
          <p className="subtitle">Daily minutes practiced</p>
        </div>
        <div className="legend">
          <span className="legend-dot practice"></span>
          <span>Practice</span>
          <span className="legend-period">Last 7 Days</span>
        </div>
      </div>

      <div className="chart-container">
        {days.map((day) => {
          const minutes = getMinutesPerDay(day.date);
          const heightPercent = maxMinutes > 0 ? (minutes / maxMinutes) * 100 : 0;
          
          return (
            <div key={day.date} className="chart-bar-container">
              <div className="chart-bar-wrapper">
                {minutes > 0 ? (
                  <div 
                    className={`chart-bar ${day.isToday ? 'today' : ''}`}
                    style={{ height: `${Math.max(heightPercent, 8)}%` }}
                  >
                    <span className="bar-tooltip">{minutes}m</span>
                  </div>
                ) : (
                  <div className="chart-bar empty">
                    <span className="bar-tooltip">-</span>
                  </div>
                )}
              </div>
              <span className={`day-label ${day.isToday ? 'today' : ''}`}>
                {day.dayName}
              </span>
            </div>
          );
        })}
      </div>

      {/* Show message if no activity at all */}
      {days.every(d => getMinutesPerDay(d.date) === 0) && (
        <div className="no-activity-message">
          No practice sessions this week yet. Time for an adventure! 🏴‍☠️
        </div>
      )}
    </div>
  );
};

export default WeeklyActivityChart;
```

```jsx
import React from 'react';

const ProgressComparison = ({ sessions, words }) => {
  const getWeekData = (weeksAgo) => {
    const now = new Date();
    const weekStart = new Date(now);
    weekStart.setDate(now.getDate() - now.getDay() - (weeksAgo * 7));
    weekStart.setHours(0, 0, 0, 0);
    
    const weekEnd = new Date(weekStart);
    weekEnd.setDate(weekStart.getDate() + 7);

    const weekSessions = sessions.filter(s => {
      const sessionDate = new Date(s.timestamp);
      return sessionDate >= weekStart && sessionDate < weekEnd;
    });

    const totalAttempts = weekSessions.reduce((sum, s) => 
      sum + (s.wordsAttempted?.length || 0), 0);
    const totalCorrect = weekSessions.reduce((sum, s) => 
      sum + (s.wordsAttempted?.filter(w => w.correct).length || 0), 0);
    
    const accuracy = totalAttempts > 0 
      ? Math.round((totalCorrect / totalAttempts) * 100) 
      : 0;

    return {
      accuracy,
      sessions: weekSessions.length,
      totalMinutes: Math.round(weekSessions.reduce((sum, s) => 
        sum + (s.durationSeconds || 0), 0) / 60),
    };
  };

  const thisWeek = getWeekData(0);
  const lastWeek = getWeekData(1);
```

```jsx
// SessionHistory.jsx

import React, { useState } from 'react';

const SessionHistory = ({ sessions }) => {
  const [expanded, setExpanded] = useState(false);

  // Sort sessions by most recent first
  const sortedSessions = [...sessions]
    .sort((a, b) => new Date(b.timestamp) - new Date(a.timestamp))
    .slice(0, expanded ? 10 : 3);

  const formatDate = (timestamp) => {
    const date = new Date(timestamp);
    return date.toLocaleDateString('en-US', { 
      weekday: 'short', 
      month: 'short', 
      day: 'numeric' 
    });
  };

  const formatTime = (timestamp) => {
    const date = new Date(timestamp);
    return date.toLocaleTimeString('en-US', { 
      hour: 'numeric', 
      minute: '2-digit' 
    });
  };

  const calculateAccuracy = (session) => {
    if (!session.wordsAttempted || session.wordsAttempted.length === 0) return 0;
    const correct = session.wordsAttempted.filter(w => w.correct).length;
    return Math.round((correct / session.wordsAttempted.length) * 100);
  };

  if (sessions.length === 0) {
    return (
      <div className="session-history">
        <h3>📅 Recent Sessions</h3>
        <p className="no-sessions">No practice sessions yet. Start an adventure!</p>
      </div>
    );
  }

  return (
    <div className="session-history">
      <h3>📅 Recent Sessions</h3>
      
      <div className="sessions-list">
        {sortedSessions.map((session, index) => {
          const accuracy = calculateAccuracy(session);
          const minutes = Math.round((session.durationSeconds || 0) / 60);
          const wordCount = session.wordsAttempted?.length || 0;

          return (
            <div key={session.id || index} className="session-item">
              <div className="session-date">
                <span className="date">{formatDate(session.timestamp)}</span>
                <span className="time">{formatTime(session.timestamp)}</span>
              </div>
              
              <div className="session-stats">
                <span className="stat">
                  <strong>{minutes}</strong> min
                </span>
                <span className="stat">
                  <strong>{wordCount}</strong> words
                </span>
                <span className={`stat accuracy ${accuracy >= 80 ? 'good' : accuracy >= 60 ? 'okay' : 'low'}`}>
                  <strong>{accuracy}%</strong>
                </span>
              </div>

              <div className="session-indicator">
                {accuracy >= 80 ? '✓' : accuracy >= 60 ? '•' : '○'}
              </div>
            </div>
          );
        })}
      </div>

      {sessions.length > 3 && (
        <button 
          className="expand-btn"
          onClick={() => setExpanded(!expanded)}
        >
          {expanded ? 'Show Less' : `Show More (${sessions.length - 3} more)`}
        </button>
      )}
    </div>
  );
};

export default SessionHistory;
```

```javascript
// analyticsCalculations.js
// COPY THIS EXACTLY - These formulas MUST be used

export const calculateAllAnalytics = (sessions, words) => {
  
  // ===== TOP METRICS =====
  
  // 1. ACCURACY - Total correct / Total attempted × 100
  const totalAttempted = sessions.reduce((sum, s) => 
    sum + (s.wordsAttempted?.length || 0), 0);
  const totalCorrect = sessions.reduce((sum, s) => 
    sum + (s.wordsAttempted?.filter(w => w.correct).length || 0), 0);
  const accuracy = totalAttempted > 0 
    ? Math.round((totalCorrect / totalAttempted) * 100) 
    : 0;

  // 2. SESSIONS - Simple count
  const totalSessions = sessions.length;

  // 3. TIME SPENT - Sum of all session durations
  const totalSeconds = sessions.reduce((sum, s) => 
    sum + (s.durationSeconds || 0), 0);
  const totalMinutes = Math.round(totalSeconds / 60);
  
  // 4. AVG SESSION - Total time / Total sessions
  const avgSessionMinutes = totalSessions > 0 
    ? Math.round((totalSeconds / 60) / totalSessions) 
    : 0;

  // 5. POINTS/TREASURES - Sum of all points earned
  const totalPoints = sessions.reduce((sum, s) => 
    sum + (s.pointsEarned || 0), 0);

  // ===== WORD MASTERY =====

  // 6. MASTERED - Words with 80%+ accuracy AND 3+ correct
  const masteredWords = words.filter(w => {
    if (w.timesAttempted < 3) return false;
    const acc = (w.timesCorrect / w.timesAttempted) * 100;
    return acc >= 80 && w.timesCorrect >= 3;
  });

  // 7. STILL PRACTICING - All other words
  const learningWords = words.filter(w => {
    if (w.timesAttempted < 3) return true;
    const acc = (w.timesCorrect / w.timesAttempted) * 100;
    return acc < 80 || w.timesCorrect < 3;
  });

  // 8. MASTERY PERCENTAGE
  const masteryPercent = words.length > 0 
    ? Math.round((masteredWords.length / words.length) * 100) 
    : 0;

  // ===== ACCURACY CHANGE =====
  
  // 9. Calculate last week's accuracy for comparison
  const oneWeekAgo = new Date();
  oneWeekAgo.setDate(oneWeekAgo.getDate() - 7);
  
  const thisWeekSessions = sessions.filter(s => 
    new Date(s.timestamp) >= oneWeekAgo);
  const lastWeekSessions = sessions.filter(s => {
    const date = new Date(s.timestamp);
    const twoWeeksAgo = new Date();
    twoWeeksAgo.setDate(twoWeeksAgo.getDate() - 14);
    return date >= twoWeeksAgo && date < oneWeekAgo;
  });

  const thisWeekAttempted = thisWeekSessions.reduce((sum, s) => 
    sum + (s.wordsAttempted?.length || 0), 0);
  const thisWeekCorrect = thisWeekSessions.reduce((sum, s) => 
    sum + (s.wordsAttempted?.filter(w => w.correct).length || 0), 0);
  const thisWeekAccuracy = thisWeekAttempted > 0 
    ? Math.round((thisWeekCorrect / thisWeekAttempted) * 100) 
    : 0;

  const lastWeekAttempted = lastWeekSessions.reduce((sum, s) => 
    sum + (s.wordsAttempted?.length || 0), 0);
  const lastWeekCorrect = lastWeekSessions.reduce((sum, s) => 
    sum + (s.wordsAttempted?.filter(w => w.correct).length || 0), 0);
  const lastWeekAccuracy = lastWeekAttempted > 0 
    ? Math.round((lastWeekCorrect / lastWeekAttempted) * 100) 
    : 0;

  const accuracyChange = thisWeekAccuracy - lastWeekAccuracy;

  // ===== STRUGGLING WORDS =====
  
  // 10. Words with <50% accuracy OR 2+ recent misses
  const strugglingWords = words.filter(w => {
    const acc = w.timesAttempted > 0 
      ? (w.timesCorrect / w.timesAttempted) * 100 
      : 0;
    return acc < 50 || (w.recentMisses || 0) >= 2;
  });

  return {
    // Top metrics
    accuracy,
    accuracyChange,
    totalSessions,
    totalMinutes,
    avgSessionMinutes,
    totalPoints,
    
    // Word mastery
    masteredCount: masteredWords.length,
    learningCount: learningWords.length,
    totalWords: words.length,
    masteryPercent,
    
    // Struggling
    strugglingWords,
    
    // Raw data for charts
    thisWeekSessions,
    lastWeekSessions,
  };
};
```
