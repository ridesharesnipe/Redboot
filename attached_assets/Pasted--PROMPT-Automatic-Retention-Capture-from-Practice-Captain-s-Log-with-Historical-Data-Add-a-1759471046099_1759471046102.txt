STEP 1: Initialize tracking at session start
```javascript
async function startPracticeSession() {
  const account = await database.getAccount();
  const childId = account.activeChildId;
  const childProfile = await database.getChildProfile(childId);
  
  // Get current week
  const currentWeek = getCurrentWeek(childProfile);
  if (!currentWeek) {
    showError("Please upload this week's spelling list first!");
    return;
  }
  
  // Initialize session tracking
  const sessionTracking = {
    sessionId: generateUniqueId('session'),
    childId: childId,
    weekId: currentWeek.weekId,
    sessionStartTime: new Date().toISOString(),
    words: currentWeek.words,
    wordTracking: {},
    missedWords: [],
    currentRound: 1,
    totalAttempts: 0
  };
  
  // Initialize tracking for each word
  currentWeek.words.forEach(word => {
    sessionTracking.wordTracking[word] = {
      firstAttemptCorrect: null,
      totalAttempts: 0,
      mastered: false,
      attemptsToMaster: null
    };
  });
  
  setState({ sessionTracking });
  
  // Start practice (existing flow continues normally)
  presentWord(currentWeek.words[0]);
}
```

STEP 2: Track each word attempt
```javascript
function handleWordAttempt(word, userInput, correctWord) {
  const isCorrect = userInput.trim().toLowerCase() === correctWord.toLowerCase();
  const { sessionTracking } = state;
  const wordData = sessionTracking.wordTracking[word];
  
  // Increment attempts
  wordData.totalAttempts++;
  sessionTracking.totalAttempts++;
  
  // Round 1: Track first attempt
  if (sessionTracking.currentRound === 1) {
    wordData.firstAttemptCorrect = isCorrect;
  }
  
  if (isCorrect) {
    // Word mastered!
    wordData.mastered = true;
    wordData.attemptsToMaster = wordData.totalAttempts;
    
    // Remove from missed words
    sessionTracking.missedWords = sessionTracking.missedWords.filter(w => w !== word);
    
    // Show feedback and advance Red Boot (existing code)
    
  } else {
    // Word wrong
    if (!sessionTracking.missedWords.includes(word)) {
      sessionTracking.missedWords.push(word);
    }
    
    // Show feedback, don't advance (existing code)
  }
  
  setState({ sessionTracking });
  
  // Continue to next word (existing code)
}
```

STEP 3: Save everything at session end
```javascript
async function endPracticeSession() {
  const { sessionTracking } = state;
  const childProfile = await database.getChildProfile(sessionTracking.childId);
  
  const currentWeek = childProfile.weeklyLists.find(
    week => week.weekId === sessionTracking.weekId
  );
  
  // ═════════════════════════════════════════════════
  // Calculate difficulty for each word
  // ═════════════════════════════════════════════════
  Object.keys(sessionTracking.wordTracking).forEach(word => {
    const wordData = sessionTracking.wordTracking[word];
    const attempts = wordData.attemptsToMaster || wordData.totalAttempts;
    
    // 1 attempt = easy, 2 = medium, 3+ = hard
    if (attempts === 1) {
      wordData.difficulty = "easy";
    } else if (attempts === 2) {
      wordData.difficulty = "medium";
    } else {
      wordData.difficulty = "hard";
    }
  });
  
  // ═════════════════════════════════════════════════
  // Save word retention data
  // ═════════════════════════════════════════════════
  if (!currentWeek.wordRetention) {
    currentWeek.wordRetention = {};
  }
  
  Object.keys(sessionTracking.wordTracking).forEach(word => {
    const wordData = sessionTracking.wordTracking[word];
    currentWeek.wordRetention[word] = {
      firstAttemptCorrect: wordData.firstAttemptCorrect,
      totalAttempts: wordData.totalAttempts,
      attemptsToMaster: wordData.attemptsToMaster,
      mastered: wordData.mastered,
      difficulty: wordData.difficulty
    };
  });
  
  // ═════════════════════════════════════════════════
  // Save practice session record
  // ═════════════════════════════════════════════════
  const duration = Math.round((Date.now() - new Date(sessionTracking.sessionStartTime)) / 1000);
  const masteredCount = Object.values(sessionTracking.wordTracking).filter(w => w.mastered).length;
  const accuracy = Math.round((masteredCount / sessionTracking.words.length) * 100);
  
  if (!currentWeek.practiceSessions) {
    currentWeek.practiceSessions = [];
  }
  
  currentWeek.practiceSessions.push({
    sessionId: sessionTracking.sessionId,
    date: new Date().toISOString(),
    duration: duration,
    totalWords: sessionTracking.words.length,
    masteredWords: masteredCount,
    totalAttempts: sessionTracking.totalAttempts,
    accuracy: accuracy
  });
  
  // ═════════════════════════════════════════════════
  // Update week summary
  // ═════════════════════════════════════════════════
  let easyCount = 0, mediumCount = 0, hardCount = 0;
  const hardWords = [];
  
  Object.keys(currentWeek.wordRetention).forEach(word => {
    const data = currentWeek.wordRetention[word];
    if (data.difficulty === "easy") easyCount++;
    else if (data.difficulty === "medium") mediumCount++;
    else if (data.difficulty === "hard") {
      hardCount++;
      hardWords.push(word);
    }
  });
  
  if (!currentWeek.weekSummary) {
    currentWeek.weekSummary = {};
  }
  
  currentWeek.weekSummary = {
    totalSessions: currentWeek.practiceSessions.length,
    totalPracticeTime: currentWeek.practiceSessions.reduce((sum, s) => sum + s.duration, 0),
    easyWords: easyCount,
    mediumWords: mediumCount,
    hardWords: hardCount,
    hardWordsList: hardWords,
    fridayTestScore: currentWeek.weekSummary?.fridayTestScore || null
  };
  
  // Save to database
  await database.updateChildProfile(sessionTracking.childId, childProfile);
  
  // Show completion screen with stats
  showSessionComplete({
    masteredWords: masteredCount,
    totalWords: sessionTracking.words.length,
    accuracy: accuracy,
    duration: duration
  });
}
```

HELPER FUNCTION:
```javascript
function getCurrentWeek(childProfile) {
  const today = new Date();
  const monday = getMonday(today);
  const weekStartDate = monday.toISOString().split('T')[0];
  
  return childProfile.weeklyLists.find(week => week.weekStartDate === weekStartDate);
}

function getMonday(date) {
  const d = new Date(date);
  const day = d.getDay();
  const diff = d.getDate() - day + (day === 0 ? -6 : 1);
  return new Date(d.setDate(diff));
}

function generateUniqueId(prefix) {
  return prefix + '_' + Date.now() + '_' + Math.random().toString(36).substr(2, 9);
}
```

function CaptainsLog() {
  const [activeChild, setActiveChild] = useState(null);
  const [weeksToShow, setWeeksToShow] = useState(5);

  useEffect(() => {
    loadActiveChild();
  }, []);

  async function loadActiveChild() {
    const account = await database.getAccount();
    const childProfile = await database.getChildProfile(account.activeChildId);
    setActiveChild(childProfile);
  }

  if (!activeChild) return <LoadingSpinner />;

  // Sort weeks newest first
  const sortedWeeks = [...activeChild.weeklyLists].sort((a, b) => 
    b.weekStartDate.localeCompare(a.weekStartDate)
  );

  return (
    <div className="captains-log">
      <button className="back-btn" onClick={() => history.back()}>
        ← Back
      </button>
      
      <h1>{activeChild.childName || "My Child"}'s Captain's Log</h1>
      <p className="grade-level">{activeChild.gradeLevel}</p>
      
      <div className="weeks-list">
        {sortedWeeks.slice(0, weeksToShow).map(week => (
          <WeekCard key={week.weekId} week={week} />
        ))}
      </div>
      
      {sortedWeeks.length > weeksToShow && (
        <button 
          className="btn-secondary"
          onClick={() => setWeeksToShow(weeksToShow + 5)}
        >
          Show More Weeks
        </button>
      )}
    </div>
  );
}

function WeekCard({ week }) {
  const [expanded, setExpanded] = useState(false);

  return (
    <div className="week-card">
      <h3>📅 Week of {formatWeekRange(week)}</h3>
      
      <div className="week-summary">
        <p className="words-preview">
          Words: {week.words.slice(0, 3).join(', ')}{week.words.length > 3 && '...'}
        </p>
        
        {week.weekSummary && (
          <>
            <div className="retention-insights">
              <h4>📊 Retention Insights:</h4>
              <p>✅ Easy Words: {week.weekSummary.easyWords}</p>
              <p>🟡 Medium Words: {week.weekSummary.mediumWords}</p>
              <p>🔴 Hard Words: {week.weekSummary.hardWords}
                {week.weekSummary.hardWordsList.length > 0 && 
                  ` (${week.weekSummary.hardWordsList.join(', ')})`
                }
              </p>
            </div>
            
            <div className="week-stats">
              <p>• Sessions: {week.weekSummary.totalSessions}</p>
              <p>• Practice Time: {Math.round(week.weekSummary.totalPracticeTime / 60)} min</p>
              {week.practiceSessions && week.practiceSessions.length > 0 && (
                <p>• Avg Accuracy: {Math.round(
                  week.practiceSessions.reduce((sum, s) => sum + s.accuracy, 0) / 
                  week.practiceSessions.length
                )}%</p>
              )}
              {week.weekSummary.fridayTestScore && (
                <p>• Test Score: {week.weekSummary.fridayTestScore}%</p>
              )}
            </div>
          </>
        )}
        
        <button 
          className="btn-link"
          onClick={() => setExpanded(!expanded)}
        >
          {expanded ? 'Hide Details' : 'View Details'}
        </button>
      </div>
      
      {expanded && <WeekDetails week={week} />}
    </div>
  );
}

function formatWeekRange(week) {
  const start = new Date(week.weekStartDate);
  const end = new Date(week.weekEndDate);
  return `${start.toLocaleDateString('en-US', {month: 'short', day: 'numeric'})} - ${end.toLocaleDateString('en-US', {month: 'short', day: 'numeric', year: 'numeric'})}`;
}

function WeekDetails({ week }) {
  return (
    <div className="week-details">
      <h4>Spelling Words:</h4>
      <div className="words-grid">
        {week.words.map((word, index) => {
          const retention = week.wordRetention?.[word];
          return (
            <div key={index} className={`word-item ${retention?.difficulty || 'unknown'}`}>
              {index + 1}. {word}
              {retention?.mastered && ' ✓'}
              {retention && (
                <span className="attempts-badge">
                  {retention.attemptsToMaster || retention.totalAttempts} tries
                </span>
              )}
            </div>
          );
        })}
      </div>
      
      {week.practiceSessions && week.practiceSessions.length > 0 && (
        <>
          <h4>Practice Sessions:</h4>
          <div className="sessions-list">
            {week.practiceSessions.map(session => (
              <div key={session.sessionId} className="session-item">
                <p className="session-date">
                  {new Date(session.date).toLocaleDateString()} - 
                  {new Date(session.date).toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})}
                </p>
                <p>• Duration: {Math.round(session.duration / 60)} minutes</p>
                <p>• Mastered: {session.masteredWords}/{session.totalWords} words</p>
                <p>• Accuracy: {session.accuracy}%</p>
              </div>
            ))}
          </div>
        </>
      )}
    </div>
  );
}
