```jsx
import React from 'react';
import './liquidGlass.css';

const BadgeCard = ({ badge, onClick }) => {
  const {
    name,
    description,
    icon,
    isUnlocked,
    rarity,
    earnedDate,
  } = badge;

  const cardClass = isUnlocked
    ? `badge-liquid-glass badge-glass-${rarity?.toLowerCase() || 'common'}`
    : 'badge-liquid-glass-locked';

  return (
    <div
      className={cardClass}
      onClick={() => onClick(badge)}
      style={{
        padding: '22px 16px 18px',
        textAlign: 'center',
        minHeight: '180px',
        display: 'flex',
        flexDirection: 'column',
        alignItems: 'center',
        justifyContent: 'center',
      }}
    >
      {/* Lock Icon - Locked Only */}
      {!isUnlocked && (
        <div className="badge-lock">🔒</div>
      )}

      {/* Icon */}
      <div className="badge-icon-wrap">
        <img src={icon} alt={name} />
      </div>

      {/* Name */}
      <div className="badge-title">{name}</div>

      {/* Description */}
      <div className="badge-desc">{description}</div>

      {/* Rarity */}
      <span className={`badge-rarity ${rarity?.toLowerCase() || 'common'}`}>
        {rarity}
      </span>

      {/* Earned Date - Unlocked Only */}
      {isUnlocked && earnedDate && (
        <div className="badge-earned">Earned {earnedDate}</div>
      )}
    </div>
  );
};

export default BadgeCard;
```

### **STEP 3: CREATE BADGE DETAIL MODAL**

**Create file: `BadgeDetailModal.jsx`**

**Copy EXACTLY:**

```jsx
import React from 'react';
import './liquidGlass.css';

const BadgeDetailModal = ({ badge, isOpen, onClose }) => {
  if (!isOpen || !badge) return null;

  const {
    name,
    description,
    icon,
    isUnlocked,
    rarity,
    earnedDate,
    progress,
    flavorText,
  } = badge;

  const modalClass = isUnlocked
    ? `badge-liquid-glass badge-glass-${rarity?.toLowerCase() || 'common'}`
    : 'badge-liquid-glass-locked';

  const progressPercent = progress
    ? Math.round((progress.current / progress.total) * 100)
    : 0;

  return (
    <>
      {/* Backdrop */}
      <div className="badge-modal-backdrop" onClick={onClose} />

      {/* Modal */}
      <div className="badge-modal-container">
        <div className={`${modalClass} badge-modal-content`}>
          
          {/* Close Button */}
          <button className="badge-modal-close" onClick={onClose}>
            ✕
          </button>

          {/* Large Icon */}
          <div className="badge-modal-icon">
            <img
              src={icon}
              alt={name}
              style={{
                filter: !isUnlocked ? 'grayscale(100%) opacity(0.5)' : 'none',
              }}
            />
          </div>

          {/* Title */}
          <h2 className="badge-modal-title">{name}</h2>

          {/* Rarity */}
          <span
            className={`badge-rarity ${rarity?.toLowerCase() || 'common'}`}
            style={{ marginBottom: '18px', display: 'inline-block' }}
          >
            {rarity}
          </span>

          {/* Description */}
          <p className="badge-modal-desc">{description}</p>

          {/* Status */}
          {isUnlocked ? (
            <div className="badge-status-unlocked">
              <div className="badge-congrats">✨ Ye Earned This Treasure! ✨</div>
              <div className="badge-date">{earnedDate}</div>
            </div>
          ) : (
            <div className="badge-status-locked">
              <div className="badge-locked-text">🔒 Locked - Keep Sailing!</div>
              {progress && (
                <>
                  <div className="badge-progress-bar">
                    <div
                      className="badge-progress-fill"
                      style={{ width: `${progressPercent}%` }}
                    />
                  </div>
                  <div className="badge-progress-text">
                    {progress.current} of {progress.total} complete
                  </div>
                </>
              )}
            </div>
          )}

          {/* Flavor Text */}
          {flavorText && (
            <p className="badge-flavor">"{flavorText}"</p>
          )}
        </div>
      </div>
    </>
  );
};

export default BadgeDetailModal;
```

### **STEP 4: UPDATE BADGE GRID/PAGE**

**Find your badges page component and update to:**

```jsx
import React, { useState } from 'react';
import BadgeCard from './BadgeCard';
import BadgeDetailModal from './BadgeDetailModal';
import './liquidGlass.css';

const BadgesPage = ({ badges }) => {
  const [selectedBadge, setSelectedBadge] = useState(null);

  return (
    <>
      {/* Badge Grid */}
      <div
        style={{
          display: 'grid',
          gridTemplateColumns: 'repeat(auto-fill, minmax(165px, 1fr))',
          gap: '16px',
          padding: '16px',
        }}
      >
        {badges.map((badge) => (
          <BadgeCard
            key={badge.id}
            badge={badge}
            onClick={(b) => setSelectedBadge(b)}
          />
        ))}
      </div>

      {/* Modal */}
      <BadgeDetailModal
        badge={selectedBadge}
        isOpen={selectedBadge !== null}
        onClose={() => setSelectedBadge(null)}
      />
    </>
  );
};

export default BadgesPage;
```
