```javascript
export const wordManager = {
  STORAGE_KEY: 'redboot-words',
  
  saveWords: (words) => {
    localStorage.setItem(wordManager.STORAGE_KEY, JSON.stringify({
      words,
      savedAt: new Date().toISOString(),
      weekNumber: Math.floor(Date.now() / (1000*60*60*24*7))
    }));
  },
  
  getWords: () => {
    const data = localStorage.getItem(wordManager.STORAGE_KEY);
    return data ? JSON.parse(data).words : [];
  },
  
  isNewWeek: () => {
    const data = localStorage.getItem(wordManager.STORAGE_KEY);
    if (!data) return true;
    const { weekNumber } = JSON.parse(data);
    const currentWeek = Math.floor(Date.now() / (1000*60*60*24*7));
    return weekNumber !== currentWeek;
  }
};
