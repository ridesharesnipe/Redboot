    <script>
        // Treasure data
        let treasures = {
            diamonds: 0,
            coins: 0,
            crowns: 0,
            bags: 0,
            pearls: 0,
            rubies: 0
        };

        // Treasure visual properties
        const treasureVisuals = {
            diamonds: { emoji: '💎', color: '#b9f2ff', size: 15 },
            coins: { emoji: '🪙', color: '#ffd700', size: 12 },
            crowns: { emoji: '👑', color: '#ffd700', size: 20 },
            bags: { emoji: '💰', color: '#8b7355', size: 18 },
            pearls: { emoji: '⚪', color: '#f8f8ff', size: 10 },
            rubies: { emoji: '🔴', color: '#dc143c', size: 14 }
        };

        // Initialize canvases
        const canvases = {};
        const contexts = {};
        
        Object.keys(treasures).forEach(type => {
            const canvas = document.getElementById(`canvas-${type}`);
            canvas.width = canvas.offsetWidth;
            canvas.height = 120;
            canvases[type] = canvas;
            contexts[type] = canvas.getContext('2d');
        });

        // Create underwater bubbles
        function createBubbles() {
            const container = document.getElementById('bubbles');
            for (let i = 0; i < 20; i++) {
                const bubble = document.createElement('div');
                bubble.className = 'bubble';
                bubble.style.left = Math.random() * 100 + '%';
                bubble.style.width = bubble.style.height = (Math.random() * 20 + 10) + 'px';
                bubble.style.animationDelay = Math.random() * 10 + 's';
                bubble.style.animationDuration = (10 + Math.random() * 10) + 's';
                container.appendChild(bubble);
            }
        }

        // Draw treasure pile on canvas
        function drawTreasurePile(type) {
            const ctx = contexts[type];
            const canvas = canvases[type];
            const count = treasures[type];
            const visual = treasureVisuals[type];
            
            // Clear canvas
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            if (count === 0) return;
            
            // Calculate grid layout
            const maxPerRow = Math.min(15, Math.ceil(Math.sqrt(count * 2)));
            const itemSize = visual.size;
            const spacing = itemSize * 0.3;
            
            // Draw each item
            let drawn = 0;
            for (let row = 0; drawn < count; row++) {
                const rowCount = Math.min(maxPerRow - Math.floor(row / 2), count - drawn);
                const startX = (canvas.width - (rowCount * (itemSize + spacing))) / 2;
                const y = canvas.height - (row + 1) * (itemSize + spacing);
                
                for (let col = 0; col < rowCount && drawn < count; col++) {
                    const x = startX + col * (itemSize + spacing);
                    
                    // Add slight randomization for natural pile look
                    const offsetX = (Math.random() - 0.5) * 4;
                    const offsetY = (Math.random() - 0.5) * 4;
                    
                    // Draw emoji
                    ctx.font = `${itemSize}px Arial`;
                    ctx.fillText(visual.emoji, x + offsetX, y + offsetY);
                    
                    drawn++;
                }
            }
        }

        // Update treasure display
        function updateTreasureDisplay(type) {
            document.getElementById(`count-${type}`).textContent = treasures[type];
            drawTreasurePile(type);
        }

        // Add treasure with animation
        function addTreasure(type, amount = 1) {
            treasures[type] += amount;
            updateTreasureDisplay(type);
            updateTotal();
            
            // Create falling coin animation
            for (let i = 0; i < Math.min(amount, 5); i++) {
                setTimeout(() => {
                    createCoinParticle(type);
                }, i * 100);
            }
            
            // Animate count
            const countEl = document.getElementById(`count-${type}`);
            countEl.style.animation = 'none';
            setTimeout(() => {
                countEl.style.animation = 'pulse 0.5s';
            }, 10);
        }

        // Create falling coin particle
        function createCoinParticle(type) {
            const particle = document.createElement('div');
            particle.className = 'coin-particle';
            particle.textContent = treasureVisuals[type].emoji;
            particle.style.left = Math.random() * window.innerWidth + 'px';
            particle.style.top = '-50px';
            document.body.appendChild(particle);
            setTimeout(() => particle.remove(), 2000);
        }

        // Update total
        function updateTotal() {
            const total = Object.values(treasures).reduce((a, b) => a + b, 0);
            document.getElementById('totalCount').textContent = total;
        }

        // Open chest animation
        function openChest() {
            const chest = document.getElementById('chest3d');
            chest.classList.add('open');
            
            // Add random treasures
            const types = Object.keys(treasures);
            const numRewards = Math.floor(Math.random() * 4) + 3;
            
            for (let i = 0; i < numRewards; i++) {
                setTimeout(() => {
                    const type = types[Math.floor(Math.random() * types.length)];
                    const amount = Math.floor(Math.random() * 10) + 5;
                    addTreasure(type, amount);
                }, i * 200);
            }
            
            setTimeout(() => {
                chest.classList.remove('open');
            }, 2000);
        }

        // Add random treasure
        function addRandomTreasure() {
            const types = Object.keys(treasures);
            const type = types[Math.floor(Math.random() * types.length)];
            const amount = Math.floor(Math.random() * 15) + 5;
            addTreasure(type, amount);
        }

        // Switch vault
        function switchVault(vault) {
            document.querySelectorAll('.vault-tab').forEach(tab => {
                tab.classList.remove('active');
            });
            event.target.classList.add('active');
            
            // Reset treasures for demo
            Object.keys(treasures).forEach(key => {
                treasures[key] = 0;
                updateTreasureDisplay(key);
            });
            updateTotal();
            
            // Change title based on vault
            const titles = {
                blackbeard: "⚓ Blackbeard's Hidden Treasure ⚓",
                sparrow: "⚓ Captain's Treasure Vault ⚓",
                davy: "⚓ Davy Jones' Sunken Fortune ⚓"
            };
            document.querySelector('.vault-title').textContent = titles[vault] || titles.sparrow;
        }

        // Initialize
        createBubbles();
        
        // Add initial treasures for demo
        setTimeout(() => {
            addTreasure('coins', 25);
            setTimeout(() => addTreasure('diamonds', 15), 500);
            setTimeout(() => addTreasure('bags', 8), 1000);
            setTimeout(() => addTreasure('crowns', 5), 1500);
        }, 1000);
    </script>
